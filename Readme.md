# Запустить все приложения в режиме разработки

npm run dev

# Запустить только API

npm run dev:api

# Запустить только Web

npm run dev:web

# Собрать все приложения

npm run build

# Запустить линтер для всех приложений

npm run lint

# Форматировать код

npm run format

# Запустить тесты

npm run test

# Проверить типы

npm run type-check

# Сгенерировать Prisma клиент

npm run db:generate

# Выполнить миграции базы данных

npm run db:migrate

# Очистить кэш и node_modules

npm run clean
